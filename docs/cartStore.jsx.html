import { create } from "zustand";

/**
 * Zustand store for managing the shopping cart.
 */
const useStoreCart = create(function (set) {
  return {
    /**
     * The array of products currently in the cart.
     * Each product should include an `id` and a `quantity` field.
     * @type {Array<Object>}
     */
    cart: [],

    /**
     * Adds a product to the cart.
     * If the product already exists, it increases its quantity by 1.
     *
     * @param {Object} product - The product object to add.
     * @param {string} product.id - The unique ID of the product.
     * @param {number} [product.quantity] - Optional existing quantity.
     */
    addToCart: function (product) {
      console.log("Added to cart: ", product.id);

      set(function (state) {
        const existingProduct = state.cart.find(function (p) {
          return p.id === product.id;
        });

        if (existingProduct) {
          return {
            cart: state.cart.map(function (p) {
              if (p.id === product.id) {
                return { ...p, quantity: (p.quantity || 1) + 1 };
              }
              return p;
            })
          };
        } else {
          return {
            cart: [...state.cart, { ...product, quantity: 1 }]
          };
        }
      });
    },

    /**
     * Removes one quantity of the specified product from the cart.
     * If quantity reaches zero, the product is removed.
     *
     * @param {string} productId - The ID of the product to remove.
     */
    removeFromCart: function (productId) {
      set(function (state) {
        return {
          cart: state.cart
            .map((p) =>
              p.id === productId ? { ...p, quantity: p.quantity - 1 } : p
            )
            .filter((p) => p.quantity > 0)
        };
      });
    },

    /**
     * Clears all items from the cart.
     */
    clearCart: function () {
      set({ cart: [] });
    }
  };
});

export default useStoreCart;
